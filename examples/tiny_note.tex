\section{A theorem}
\begin{theorem}[Euler]
For every real number $x$,
\[
\exp\left(i \cdot \pi\right) + 1 = 0
\]
\end{theorem}
\begin{proof}
Use the complex exponential definition.
\[
\exp\left(i \cdot x\right) = \cos\left(x\right) + i \cdot \sin\left(x\right)
\]
Set $x = \pi$.
\end{proof}

\section{Mathematics}
Inline math works naturally: $x^2 + y^2 = z^2$.
Custom macros compile to ordinary LaTeX: $\norm{v} = \sqrt{v \cdot v}$.
\begin{equation}
\label{gaussian}
\int_{-\infty}^{\infty} \exp\left(-x^2\right) \, dx = \sqrt{\pi}
\end{equation}
Equation \ref{gaussian} is the classic Gaussian integral.
\[
\sum_{n=1}^{\infty} \frac{1}{n^2} = \pi^2 / 6
\]
